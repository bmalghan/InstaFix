class User
{
}

class Administrator
{
  isA User;
  1 -- 1 Login;
  1 -- * Profile;
}

class Client
{
  isA User;
  * -- 1 Login;
  * -- 1 CreateAccount;
}

class ServiceProvider
{
  isA User;
  * -- 1 Login;
}

class Login
{
  username;
  password;
  1 -- 1 Welcome;
}

class CreateAccount
{
  1 -- * ServiceProvider;
  1 -- 1 Profile;
  1 -- 1 Welcome;
}

class Profile
{
  firstname;
  lastname;
  username;
  email;
  password;
  Bool client;
  Bool serviceProvider;
}

class Welcome
{
}
